    Class BaseIntent

    abstract base class for a separate intent

    Inherits from

    • Object (base class)

    Constructors

    Name Description
    this c'tor

    Properties

    Name Type Description
    name [get] string allows to query for the intents string representation that needs to match intent schema
    textManager [set] ITextManager allows to define used textManager

    Methods

    Name Description
    onIntent handler that needs to be implemented in inheriting Intent
    getText forwards to currently active ITextManager.getText

    Example

    class TestIntent : BaseIntent{
	override AlexaResult onIntent(AlexaEvent, AlexaContext){
		AlexaResult res;
		res._version = "v3";
		return res;
	}
}

class TestSkill : AlexaSkill!TestSkill{
	this(){
		super([]);
		addIntent(new TestIntent);
	}
}

auto skill = new TestSkill();
AlexaEvent ev;
ev.request.type = AlexaRequest.Type.IntentRequest;
ev.request.intent.name = "TestIntent";
auto res = skill.executeEvent(ev,AlexaContext());
assert(res._version == "v3");
